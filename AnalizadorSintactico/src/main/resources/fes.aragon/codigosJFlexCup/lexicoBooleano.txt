package fes.aragon.lexico;

%%
%class LexicoBooleano
%line
%column
%type TokensBooleano

espacio=[ \t\r\n]+

%{
    public String lexema;
    public int getYyline(){
        return yyline+1;
    }
    public int getYycolumn(){
        return yycolumn;
    }
%}

%%
{espacio} {/* Ignora espacios en blanco */}

"true"               { lexema=yytext(); return TokensBooleano.TRUE; }
"false"              { lexema=yytext(); return TokensBooleano.FALSE; }
"and"                { lexema=yytext(); return TokensBooleano.AND; }
"or"                 { lexema=yytext(); return TokensBooleano.OR; }
"not"                { lexema=yytext(); return TokensBooleano.NOT; }

"("                  { lexema=yytext(); return TokensBooleano.PARENTESIS_A; }
")"                  { lexema=yytext(); return TokensBooleano.PARENTESIS_C; }
";"                  { lexema=yytext(); return TokensBooleano.PUNTO_COMA; }

.                    { lexema=yytext(); return TokensBooleano.ERROR; }
